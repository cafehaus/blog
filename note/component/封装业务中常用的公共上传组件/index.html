# е°ҒиЈ…дёҡеҠЎдёӯеёёз”Ёзҡ„е…¬е…ұдёҠдј з»„д»¶

зӣҙжҺҘз”Ё input зҡ„ file еұһжҖ§пјҢеӣҫзүҮеӣһжҳҫиҰҒз”ЁеҲ° FileReader иҜ»еҸ–ж–Үд»¶жҲҗ base64пјҢеҺӢзј©еӣҫзүҮеҸҜд»ҘеҲ©з”Ё canvasпјҢеҰӮжһңиҰҒзј“еӯҳеңЁжң¬ең°з”ұдәҺж–Үд»¶дёҖиҲ¬йғҪиҫғеӨ§пјҢlocalStorage д№Ӣзұ»зҡ„жң¬ең°еӯҳеӮЁйғҪжңүеӨ§е°ҸйҷҗеҲ¶дёҚеҗҲйҖӮпјҢеҸҜд»Ҙз”Ё indexedDB жң¬ең°ж•°жҚ®еә“еҺ»е®һзҺ°гҖӮ

жіЁж„ҸжӢ–жӢҪдёҠдј  @drop йңҖиҰҒйҳ»жӯўдёҖдёӘй»ҳи®ӨдәӢд»¶ @dragover.prevent жүҚиғҪз”ҹж•ҲгҖӮ

<template>
  <div
    class="v-upload"
    @dragover.prevent
    @drop="handleDrop"
    @click="handleUpload"
  >
    <slot>
      <button>дёҠдј </button>
    </slot>

    <input
      ref="fileInput"
      type="file"
      style="display: none;"
      :accept="accept"
      @change="handleChange"
    >
  </div>
</template>

<script>
  /**
   * е…¬е…ұдёҠдј з»„д»¶
   * дёҡеҠЎдёӯеҚ•дёӘдёҠдј еӣҫзүҮе’Ңexcelж–Үд»¶иҫғеӨҡпјҢж ·ејҸеҸҜзӣҙжҺҘдәӨз»ҷзҲ¶з»„д»¶жҺ§еҲ¶пјҢдёҠдј з»„д»¶еҶ…еҸӘеҒҡйҖүжӢ©ж–Үд»¶зҡ„зӣёе…ійҖ»иҫ‘
   * ie е·Іжӯ»пјҢдёҚз”ЁиҖғиҷ‘ ie дәҶ
   * TODOпјҡ1. жү©еұ•еӨҡйҖүж–Үд»¶пјҢдёҚиҝҮиҝҷдёӘеҠҹиғҪдёҡеҠЎдёӯдҪҝз”Ёиҫғе°‘пјҢеӨҡдёӘж–Үд»¶еҸҜеӨҡж¬Ўи°ғз”ЁжӯӨз»„д»¶е®һзҺ°
   *       2. зӣҙжҺҘи°ғз”Ёе…¬е…ұдёҠдј жҺҘеҸЈдёҠдј еҲ° CDN дёҠпјҢзӣҙжҺҘ return иҺ·еҸ–еҲ°зҡ„ CDN иө„жәҗй“ҫжҺҘ
   */
  export default {
    name: 'VUpload',
    props: {
      drag: { // жҳҜеҗҰеҸҜжӢ–жӢҪдёҠдј 
        type: Boolean,
        default: false,
      },
      maxSize: { // ж–Үд»¶еӨ§е°ҸйҷҗеҲ¶пјҢеҚ•дҪҚ MпјҢй»ҳи®ӨжңҖеӨ§ 2 MпјҢдёә 0 еҲҷдёҚйҷҗеҲ¶еӨ§е°Ҹ
        type: Number,
        default: 0,
      },
      accept: { // е…Ғи®ёзҡ„ж–Үд»¶ж јејҸпјҢinput ж ҮзӯҫеҺҹз”ҹзҡ„еұһжҖ§пјҢдјҡеңЁйҖүжӢ©ж–Үд»¶ж—¶иҝҮж»Ө
        type: String,
        default: '', // image/*  video/*  audio/*  .pdf  text/html ...
      },
      format: { // ж”ҜжҢҒзҡ„ж–Үд»¶зұ»еһӢпјҢиҜҶеҲ«ж–Үд»¶зҡ„еҗҺзјҖеҗҚпјҢеҸҜдёҺ accept еұһжҖ§з»“еҗҲдҪҝз”Ё
        type: Array,
        default: () => [],
      },
      compress: { // еҺӢзј©йҷҗеҲ¶пјҢй»ҳи®Ө 1 MпјҢдёҠдј еӣҫзүҮж—¶и¶…иҝҮжӯӨеӨ§е°ҸиҮӘеҠЁеҺӢзј©пјҢдёә 0 еҲҷдёҚеҺӢзј©
        type: Number,
        default: 1
      },
      // multiple: { // жҳҜеҗҰж”ҜжҢҒеӨҡйҖүж–Үд»¶
      //   type: Boolean,
      //   default: false
      // },
      // uploadServer: { // жҳҜеҗҰзӣҙжҺҘдёҠдј еҲ°жңҚеҠЎеҷЁпјҢеҸҜй…ҚеҗҲ api дҪҝз”Ё
      //   type: Boolean,
      //   default: false
      // },
      // api: { // дёҠдј жҺҘеҸЈең°еқҖ
      //   type: String,
      //   default: '',
      // }
    },
    data() {
      return {}
    },
    methods: {
      handleUpload() {
        this.$refs.fileInput.dispatchEvent(new MouseEvent('click'))
      },

      /**
       * жӢ–жӢҪ
       */
      handleDrop(e) {
        e.preventDefault()
        if (!this.drag) return
        const files = e.dataTransfer.files
        if (!files.length) return
        this.handleChange(files, true)
      },

      handleChange(f, isDrop) {
        const files = (isDrop && f) ? f : this.$refs.fileInput.files
        if (files.length > 0) {
          const file = files[0]
          const fileSize = file.size
          const fileName = file.name
          const fileSuffix = fileName.split('.').pop().toLocaleLowerCase() // иҪ¬жҚўжҲҗе°ҸеҶҷеҗҺзјҖпјҢжіЁж„ҸдёҚеҢ…жӢ¬зӮ№ .
          // const imgSuffix = `(bmp|jpg|png|tif|gif|pcx|tga|exif|fpx|svg|psd|cdr|pcd|dxf|ufo|eps|ai|raw|WMF|webp|jpeg)`

          // еҸӘеҢ№й…Қ img ж ҮзӯҫеҸҜд»ҘзӣҙжҺҘжҳҫзӨәзҡ„еӣҫзүҮ
          // https://developer.mozilla.org/zh-CN/docs/Web/HTML/Element/img
          // https://developer.mozilla.org/zh-CN/docs/Web/Media/Formats/Image_types
          const imgSuffix = '(apng|avif|bmp|gif|ico|jpeg|jpg|png|svg|tiff|webp)'
          // const imgReg = new RegExp(`.*\.${imgSuffix}$`)
          const imgReg = new RegExp(`${imgSuffix}$`)
          const isImg = imgReg.test(fileSuffix)
          const fileFormatName = isImg ? 'еӣҫзүҮ' : 'ж–Үд»¶'

          // ж–Үд»¶еӨ§е°Ҹж ЎйӘҢ
          const n = this.maxSize
          if ((n > 0) && fileSize > 1024 * 1024 * n) {
            this.$message.warning(`${fileFormatName}еӨ§е°ҸдёҚиғҪи¶…иҝҮ${n}MпјҢиҜ·еҺӢзј©еҗҺдёҠдј пјҒ`)
            this.$refs.fileInput.value = ''
            return
          }

          // ж–Үд»¶ж јејҸж ЎйӘҢ
          if (this.format.length) {
            const isValid = this.format.some(item => item.toLocaleLowerCase() === fileSuffix)
            if (!isValid) {
              this.$message.warning(`${fileFormatName}ж јејҸдёҚжӯЈзЎ®пјҒ`)
              this.$refs.fileInput.value = ''
              return
            }
          }

          this.readFile({ file, fileName, fileSize, isImg })
        } else {
          this.$message.error('иҜ·еҲ·ж–°йЎөйқўеҗҺйҮҚж–°дёҠдј пјҒ')
        }
        this.$refs.fileInput.value = ''
      },

      /**
       * иҜ»еҸ–ж–Үд»¶дҝЎжҒҜ
       */
      readFile({ file, fileName, fileSize, isImg }) {
        const reader = new FileReader()
        reader.readAsDataURL(file)
        reader.onload = e => {
          // иҜ»еҸ–ж–Үд»¶жң¬ең°ең°еқҖпјҢз”ЁеңЁйЎөйқўдёҠеӣһжҳҫ
          const fileUrl = e.target.result

          if (!isImg || (this.compress <= 0 || fileSize <= this.compress * 1024 * 1024)) { // дёҚеҺӢзј©
            this.handleEmit({ file, fileName, fileSize, isImg, fileUrl })
          } else {
            this.compressImage({ file, fileName, fileSize, fileUrl, isImg })
          }
        }
        reader.onerror = e => {
          console.error(e)
        }
      },

      /**
       * еҺӢзј©еӣҫзүҮ
       */
      compressImage({ file, fileName, fileSize, isImg, fileUrl }) {
        const img = new Image()
        img.src = fileUrl
        img.onload = e => {
          const w = img.width / 1.5
          const h = img.height / 1.5
          const quality = 0.7 // й»ҳи®ӨеӣҫзүҮиҙЁйҮҸдёә0.92
          // з”ҹжҲҗcanvas
          const canvas = document.createElement('canvas')
          const ctx = canvas.getContext('2d')
          // еҲӣе»әеұһжҖ§иҠӮзӮ№
          const anw = document.createAttribute('width')
          anw.nodeValue = w
          const anh = document.createAttribute('height')
          anh.nodeValue = h
          canvas.setAttributeNode(anw)
          canvas.setAttributeNode(anh)

          // й“әеә•иүІ PNGиҪ¬JPEGж—¶йҖҸжҳҺеҢәеҹҹдјҡеҸҳй»‘иүІ
          ctx.fillStyle = '#fff'
          ctx.fillRect(0, 0, w, h)

          ctx.drawImage(img, 0, 0, w, h)
          // qualityеҖји¶Ҡе°ҸпјҢжүҖз»ҳеҲ¶еҮәзҡ„еӣҫеғҸи¶ҠжЁЎзіҠ
          const base64 = canvas.toDataURL('image/jpeg', quality) // еӣҫзүҮж јејҸjpegжҲ–webpеҸҜд»ҘйҖү0-1иҙЁйҮҸеҢәй—ҙ

          // еҺ»жҺүurlзҡ„еӨҙпјҢе№¶иҪ¬жҚўдёәbyte
          const bytes = window.atob(base64.split(',')[1])
          // еӨ„зҗҶејӮеёё,е°Ҷasciiз Ғе°ҸдәҺ0зҡ„иҪ¬жҚўдёәеӨ§дәҺ0
          const ab = new ArrayBuffer(bytes.length)
          const ia = new Uint8Array(ab)
          for (let i = 0; i < bytes.length; i++) {
            ia[i] = bytes.charCodeAt(i)
          }
          let newFile = new File([ab], fileName, { type: 'image/jpeg' })
          this.handleEmit({ file: newFile, fileName, fileSize, fileUrl, isImg })
        }
        img.onerror = e => {
          console.error(e)
        }
      },

      /**
       * е°Ҷж–Үд»¶дҝЎжҒҜдј йҖ’з»ҷзҲ¶з»„д»¶
       */
      handleEmit({ file, fileName, fileSize, fileUrl, isImg }) {
        this.$emit('file', { file, fileName, fileSize, fileUrl, isImg })
      }
    },
  }
</script>
